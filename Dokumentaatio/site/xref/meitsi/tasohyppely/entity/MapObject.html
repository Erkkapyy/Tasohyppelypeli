54      public MapObject(TileMap tm) {
55          tileMap = tm;
56          tileSize = tm.getTileSize();
57      }
58  
59      public boolean intersects(MapObject o) {
60          Rectangle r1 = getRectangle();
61          Rectangle r2 = o.getRectangle();
62          return r1.intersects(r2);
63      }
64  
65      public Rectangle getRectangle() {
66          return new Rectangle((int) x - cwidth, (int) y - cheight, cwidth, cheight);
67      }
68  
69      public void calculateCorners(double x, double y) {
70          int leftTile = (int) (x - cwidth / 2) / tileSize;
71          int rightTile = (int) (x + cwidth / 2 - 1) / tileSize;
72          int topTile = (int) (y - cheight / 2) / tileSize;
73          int bottomTile = (int) (y + cheight / 2 - 1) / tileSize;
74          if (topTile < 0 || bottomTile >= tileMap.getNumRows() || leftTile < 0 || rightTile >= tileMap.getNumCols()) {
75              topLeft = topRight = bottomLeft = bottomRight = false;
76              return;
77          }
78          int tl = tileMap.getType(topTile, leftTile);
79          int tr = tileMap.getType(topTile, rightTile);
80          int bl = tileMap.getType(bottomTile, leftTile);
81          int br = tileMap.getType(bottomTile, rightTile);
82          topLeft = tl == Tile.BLOCKED;
83          topRight = tr == Tile.BLOCKED;
84          bottomLeft = bl == Tile.BLOCKED;
85          bottomRight = br == Tile.BLOCKED;
86  
87      }
88  
89      public void checkTileMapCollision() {
90          currCol = (int) x / tileSize;
91          currRow = (int) y / tileSize;
92          xdest = x + dx;
93          ydest = y + dy;
94  
95          xtemp = x;
96          ytemp = y;
97  
98          calculateCorners(x, ydest);
99          if (dy < 0) {
100             if (topLeft || topRight) {
101                 dy = 0;
102                 ytemp = currRow * tileSize + cheight / 2;
103             }
104         } else {
105             ytemp += dy;
106         }
107         if (dy > 0) {
108             if (bottomLeft || bottomRight) {
109                 dy = 0;
110                 falling = false;
111                 ytemp = (currRow + 1) * tileSize - cheight / 2;
112             } else {
113                 ytemp += dy;
114             }
115         }
116 
117         calculateCorners(xdest, y);
118         if (dx < 0) {
119             if (topLeft || bottomLeft) {
120                 dx = 0;
121                 xtemp = currCol * tileSize + cwidth / 2;
122             }
123         } else {
124             xtemp += dx;
125         }
126         if (dx > 0) {
127             if (topRight || bottomRight) {
128                 dx = 0;
129                 xtemp = (currCol + 1) * tileSize - cwidth / 2;
130             } else {
131                 ytemp += dy;
132             }
133         }
134 
135         if (!falling) {
136             calculateCorners(x, ydest + 1);
137             if (!bottomLeft && !bottomRight) {
138                 falling = true;
139             }
140         }
141 
142     }
